## 코드/실험 & 깨달은 점

### 1. **함수 + return 구조 실습 / 오차 테스트**

```js
function returnTest() {
	console.log('1');
	return 2;
	console.log('이거 안보임'); // 실행 안됨!
}

function vat2(a) {
	return a * 1.1;
}
console.log(vat2(55555)); // 61110.50000000001 (오차 발생)
```

- return 이후 코드는 실행되지 않는다!
- 1.1 \* 55555 등에서 JS의 소수점 오차를 체험.

### 2. **할인 함수 구현 & 소수변환 실험**

```js
function 할인가격변환기(price, first고객) {
	let sale = (price * 0.9).toFixed(1);
	let 숫자변환기 = parseFloat(sale);
	if (first고객 === true) {
		return Number((숫자변환기 - 1.5).toFixed(1));
	} else {
		return 숫자변환기;
	}
}
console.log(할인가격변환기(70, false)); // 63
console.log(할인가격변환기(10, true)); // 7.5
```

- toFixed()는 문자열을 만들어 반환 → 숫자로 연산 시 parseFloat/Number 꼭 사용
- 소수 실수 오차는 반올림 등으로 처리해주는 습관 들이기

### 3. **스크롤 이벤트 실험 & DOM 속성 활용**

스크롤 이벤트(window, div)에 각각 실습

```js
// 전체 스크롤 위치(px) 확인
window.addEventListener('scroll', function () {
	console.log(window.scrollY);
});

// div 스크롤, 약관 박스 끝까지 내릴 때 경고창
document.querySelector('.lorem').addEventListener('scroll', function () {
	let 스크롤양 = this.scrollTop;
	let 실제높이 = this.scrollHeight;
	let 디브높이 = this.clientHeight;
	// 눈에 보이는 영역 + 내린양 = 실제높이 에 도달했을 때
	if (스크롤양 + 디브높이 >= 실제높이) {
		alert('약관 전체 확인!');
	}
});
```

- window/element 각각 scroll 이벤트, Y축 위치 확인법
- div는 scrollTop+clientHeight==scrollHeight로 '거의 끝' 판단
- clientHeight, scrollHeight, scrollTop 각각 값 차이 실습
